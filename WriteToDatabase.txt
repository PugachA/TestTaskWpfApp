public void WriteToDatabase(string SqlServer, string Database, DataXmlSql Data)
        {
            try
            {
                string connectionString = $@"Data Source=.\{SqlServer};Initial Catalog={Database};Integrated Security=True"; //подключаемся к базе

                using (SqlConnection connection = new SqlConnection(connectionString))
                {
                    connection.Open();

                    for (int i = 0; i < Data.value.GetUpperBound(0) + 1; i++)
                    {
                        SqlCommand command = new SqlCommand(Data.sqlStoredProcedure, connection);
                        // указываем, что команда представляет хранимую процедуру
                        command.CommandType = System.Data.CommandType.StoredProcedure;

                        for (int j = 0; j < Data.value.GetUpperBound(1) + 1; j++)
                        {
                            SqlParameter Param = new SqlParameter
                            {
                                ParameterName = Data.parametersStoredProcedure[j],
                                Value = Data.value[i, j]
                            };
                            //добавляем параметр
                            command.Parameters.Add(Param);
                        }
                        var result = command.ExecuteNonQuery(); //результат количество добавленных или обновленных строк
                    }
                }
                logger.Info($"{Data.name} успешно записан в БД");
            }
            catch (Exception ex)
            {
                logger.Error(ex.Message);
            }
        }
